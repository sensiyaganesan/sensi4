class Solution {

    /**
     * @param Integer $num
     * @return Integer
     */
    function maximum69Number($num)
    {
        $numStr = (string) $num;
        $index = strpos($numStr, '6');
        if ($index !== false)
         {
            $numStr[$index] = '9';
        }
        return (int) $numStr;
    }
}
